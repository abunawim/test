Schema::create('articles', function (Blueprint $table) {
    $table->increments('id');
	$table->integer('category_id');
    $table->integer('user_id');
    $table->string('title');
	$table->string('slug')->unique();
    $table->text('content');
    $table->string('file')->nullable();
    $table->softDeletes();
    $table->timestamps();
});
