CREATE TABLE NM_SEMESTER
(
  SEMESTER_ID       VARCHAR2(16 BYTE),
  SEMESTER_NUMBER   NUMBER,
  SEMESTER_NAME     VARCHAR2(100 BYTE)          NOT NULL,
  REMARKS           VARCHAR2(500 BYTE),
  ACTIVE_FLAG       VARCHAR2(1 BYTE),
  ENTERED_BY        VARCHAR2(16 BYTE),
  ENTRY_TIMESTAMP   DATE,
  UPDATED_BY        VARCHAR2(16 BYTE),
  UPDATE_TIMESTAMP  DATE,
  COMPANY_NO        VARCHAR2(10 BYTE)           DEFAULT 1
);

COMMENT ON COLUMN NM_SEMESTER.SEMESTER_NUMBER IS 'SEMESTER_NUMBER is number of semester Ex. 3 Semester is 3';


ALTER TABLE NM_SEMESTER ADD (
  CONSTRAINT PK_NM_SEMESTER
  PRIMARY KEY  (SEMESTER_ID));
