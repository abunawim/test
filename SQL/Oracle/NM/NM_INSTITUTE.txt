CREATE TABLE NM_INSTITUTE
(
  INSTITUTE_ID      VARCHAR2(16 BYTE),
  INSTITUTE_UDID    VARCHAR2(16 BYTE),
  INSTITUTE_NAME    VARCHAR2(100 BYTE)          NOT NULL,
  REMARKS           VARCHAR2(500 BYTE),
  ACTIVE_FLAG       VARCHAR2(1 BYTE),
  ENTERED_BY        VARCHAR2(16 BYTE),
  ENTRY_TIMESTAMP   DATE,
  UPDATED_BY        VARCHAR2(16 BYTE),
  UPDATE_TIMESTAMP  DATE,
  COMPANY_NO        VARCHAR2(10 BYTE)           DEFAULT 1
);


ALTER TABLE NM_INSTITUTE ADD (
  CONSTRAINT PK_NM_INSTITUTE
  PRIMARY KEY  (INSTITUTE_ID));
