CREATE TABLE NM_BOARD
(

  BOARD_UDID        VARCHAR2(16 BYTE),
  BOARD_NAME        VARCHAR2(100 BYTE)          NOT NULL,
  REMARKS           VARCHAR2(500 BYTE),
