CREATE TABLE NM_OCCUPATION
(
  OCCUPATION_ID        VARCHAR2(16 BYTE),
  OCCUPATION_NAME_ENG  VARCHAR2(100 BYTE)       NOT NULL,
  OCCUPATION_NAME_BAN  VARCHAR2(100 BYTE),
  REMARKS              VARCHAR2(500 BYTE),
  ACTIVE_FLAG          VARCHAR2(1 BYTE),
  ENTERED_BY           VARCHAR2(16 BYTE),
  ENTRY_TIMESTAMP      DATE,
  UPDATED_BY           VARCHAR2(16 BYTE),
  UPDATE_TIMESTAMP     DATE,
  COMPANY_NO           VARCHAR2(10 BYTE)        DEFAULT 1
);


ALTER TABLE NM_OCCUPATION ADD (
  CONSTRAINT PK_NM_OCCUPATION
  PRIMARY KEY  (OCCUPATION_ID));
