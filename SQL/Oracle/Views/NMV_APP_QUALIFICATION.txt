
  CREATE OR REPLACE FORCE VIEW "KYAMC_MCMS"."NMV_APP_QUALIFICATION" ("REG_QUALI_ID", "APPLICATION_ID", "SL_NO", "EXAM_ID", "EXAM_NAME", "PASSING_YEAR", "BOARD_ID", "BOARD_NAME", "EXAMGROUP_ID", "EXAMGROUP_NAME", "EXAMSUBJECT_ID", "EXAMSUBJECT_NAME", "INSTITUTE_ID", "INSTITUTE_NAME", "GPA", "REG_NO") AS 
  SELECT
 Q.REG_QUALI_ID,
 Q.APPLICATION_ID,
 Q.SL_NO,
 Q.EXAM_ID,
 (SELECT E.EXAM_NAME FROM NM_EXAM E WHERE  Q.EXAM_ID=E.EXAM_ID) EXAM_NAME,
 Q.PASSING_YEAR,
 Q.BOARD_ID,
 (SELECT B.BOARD_NAME FROM NM_BOARD B WHERE  Q.BOARD_ID=B.BOARD_ID) BOARD_NAME,
 Q.EXAMGROUP_ID,
 (SELECT G.EXAMGROUP_NAME FROM NM_EXAMGROUP G WHERE  Q.EXAMGROUP_ID=G.EXAMGROUP_ID) EXAMGROUP_NAME,
 Q.EXAMSUBJECT_ID,
 (SELECT S.EXAMSUBJECT_NAME FROM NM_EXAMSUBJECT S WHERE  Q.EXAMSUBJECT_ID=S.EXAMSUBJECT_ID) EXAMSUBJECT_NAME,
 Q.INSTITUTE_ID,
 Q.INSTITUTE_NAME,
 Q.GPA,
 Q.REG_NO
FROM NM_APP_QUALIFICATION Q;
