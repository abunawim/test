  A.SESSION_ID,
  (SELECT S.SESSION_NAME FROM NM_SESSION S WHERE A.SESSION_ID=S.SESSION_ID) SESSION_NAME,
  A.APPLICATION_SL,
  A.COURSE_ID,
  (SELECT C.COURSE_NAME FROM NM_COURSE C WHERE A.COURSE_ID=C.COURSE_ID) COURSE_NAME,

  (SELECT B.BATCH_NAME FROM NM_BATCH_MAP B WHERE A.BATCH_MAP_ID=B.BATCH_MAP_ID) BATCH_NAME,
  A.POLICYMST_ID,
  (SELECT P.POLICYMST_NAME FROM NM_POLICYMST P WHERE A.POLICYMST_ID=P.POLICYMST_ID) POLICYMST_NAME,
  A.FULL_NAME_ENG,
  A.FULL_NAME_BNG,
  A.BIRTH_DATE,

  A.FATHER_EMAIL,

  (SELECT O.OCCUPATION_NAME_ENG FROM NM_OCCUPATION O WHERE A.MOTHER_OCCU_ID=O.OCCUPATION_ID) MOTHER_OCCUPATION,
  A.GUARDIAN_SIGN,
  A.GUARDIAN_SIGN_DATE,
  A.LG_NAME,
  A.LG_DETAILS,
  A.LG_OCCU_ID,
  (SELECT O.OCCUPATION_NAME_ENG FROM NM_OCCUPATION O WHERE A.LG_OCCU_ID=O.OCCUPATION_ID) LG_OCCUPATION,
  A.LG_MOBILE,
  A.LG_EMAIL,
  A.LG_VILLAGE,
  A.LG_POST_OFFICE,
  A.LG_UPAZILLA,
  A.LG_DISTRICT,

  A.UPDATED_BY,
  A.UPDATE_TIMESTAMP,
  A.COMPANY_NO,

  A.APPLICANT_SIGN_PATH,
  A.ADDRESS_TYPE,
  A.LOCAL_GUARDIAN_FG,
  A.QUOTA_ID,
  (SELECT Q.QUOTA_NAME FROM NM_QUOTA Q WHERE A.QUOTA_ID=Q.QUOTA_ID) QUOTA_NAME
